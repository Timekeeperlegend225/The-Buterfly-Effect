REPLACE:achievement_be_prepared = {
	possible = {
		always = yes
	}
	happened = {
		custom_tooltip = {
			text = achievement_be_prepared_tt
			any_scope_state = {
				any_harvest_condition = {
					harvest_condition_intensity = { target = prev.owner value > 5 }
				}
				state_region = {
					NOR = {
						has_harvest_condition = pollinator_surge
						has_harvest_condition = optimal_sunlight
						has_harvest_condition = moderate_rainfall
					}
				}
			}
		}
		custom_tooltip = {
			text = achievement_be_prepared_2_tt
			NOT = {
				any_scope_state = {
					has_famine = yes
				}
			}
		}
		custom_tooltip = {
			text = achievement_be_prepared_3_tt
			OR = {
				AND = {
					institution_investment_level = {
						institution = institution_home_affairs
						value >= 6
					}
					has_law_or_variant = law_type:law_national_guard
				}
				AND = {
					institution_investment_level = {
						institution = institution_health_system
						value >= 6
					}
					has_law_or_variant = law_type:law_charitable_health_system
				}
				AND = {
					institution_investment_level = {
						institution = institution_social_security
						value >= 6
					}
				}
			}
		}
	}
}
