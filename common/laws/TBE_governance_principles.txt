# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

REPLACE:law_monarchy = {
	group = lawgroup_governance_principles

	icon = "gfx/interface/icons/law_icons/monarchy.dds"

	progressiveness = 0

	modifier = {
		country_head_of_state_popularity_legitimacy_add = 10
	}

	ai_will_do = {
		exists = ruler
		ruler = {
			has_monarchist_ideology = yes
		}
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_monarchy
				has_strategy = ai_strategy_reactionary_agenda
				power_bloc ?= { has_identity = identity:identity_ideological_union }
				scope:target_country = { # Is their law different enough from what we want to impose?
					NOT = { has_law_or_variant = law_type:law_theocracy }
				}
			}

			add = base_impose_law_weight
		}

		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= this
					OR = {
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}
}

REPLACE_OR_CREATE:law_institutional_rule = {
	group = lawgroup_governance_principles

	icon = "gfx/interface/icons/law_icons/theocracy.dds"

	progressiveness = -25

	modifier = {
		country_institutional_ig_in_gov_legitimacy_add = 20
		country_legitimacy_base_add = -10
	}

	on_activate = {
		custom_tooltip = "The law will get an amendment dependent of your instituional IG"
		random_law = {
			limit = {
				type = law_type:law_institutional_rule
			}
			save_scope_as = institutional_rule_scope
		}
		TBE_insitutional_rule_initial_amendment = yes
	}

	ai_will_do = {
		exists = ruler
		OR = {
			ruler = {
				has_ideology = ideology:ideology_theocrat
			}
			any_interest_group = {
				is_in_government = yes
				is_powerful = yes
				leader = {
					has_ideology = ideology:ideology_theocrat
				}
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_theocracy
				has_strategy = ai_strategy_reactionary_agenda
				power_bloc ?= { has_identity = identity:identity_religious }
				religion = scope:target_country.religion
				scope:target_country = { # Is their law different enough from what we want to impose?
					NOT = { has_law = law_type:law_monarchy }
				}
			}

			add = base_impose_law_weight
			if = {
				limit = {
					scope:target_country = { has_law_or_variant = law_type:law_council_republic }
					NOT = { has_strategy = ai_strategy_egalitarian_agenda }
				}
				multiply = 5
			}
			else = {
				multiply = 0.5 # AI is less likely to mess with governance principles unless dealing with council republics
			}
		}

		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= this
					OR = {
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}
}

		democracy

		OR = {
			AND = {
