REPLACE:law_autocracy = {
	group = lawgroup_legislative_principle

	icon = "gfx/interface/icons/law_icons/autocracy.dds"

	progressiveness = -100

	modifier = {
		country_authority_add = 300
		country_legitimacy_ideological_incoherence_mult = 0.5
		country_officers_pol_str_mult = 0.5
	}

	disallowing_laws = {
		law_head_of_legislature
	}

	on_activate = {
		scripted_effect_parties_disappearence = yes
		cleanup_voting_strength_modifiers = yes
		if = {
			limit = {
				has_law_or_variant = law_type:law_aristocratic
			}
			activate_law = law_type:law_divine_right
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_institutional
			}
			activate_law = law_type:law_loyalism
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_plutocratic
			}
			activate_law = law_type:law_clientelism
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_meritocratic
			}
			activate_law = law_type:law_enlightened_despotate
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_egalitarian
			}
			activate_law = law_type:law_populism
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_partisan
			}
			activate_law = law_type:law_personality_rule
		}
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_autocracy
				OR = {
					AND = {
						power_bloc ?= { has_identity = identity:identity_ideological_union }
						has_strategy = ai_strategy_conservative_agenda
					}
					has_strategy = ai_strategy_reactionary_agenda
				}
				scope:target_country = {
					NOR = { # Is their law different enough from what we want to impose?
						has_law_or_variant = law_type:law_oligarchy
						has_law_or_variant = law_type:law_aristocratic
					}
				}
			}

			add = base_impose_law_weight
		}
	}
}

		democracy
	}

		country_authority_add = 200
		country_legitimacy_ideological_incoherence_mult = -0.1
	}

	on_activate = {
		if = {
			limit = {
				game_date > 1836.1.1

		}

			limit = {

		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_institutional
			}
			activate_law = law_type:law_institutional_council
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_plutocratic
			}
			activate_law = law_type:law_company_council
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_meritocratic
			}
			activate_law = law_type:law_supervised_participation
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_egalitarian
			}
			activate_law = law_type:law_representative_council
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_partisan
			}
			activate_law = law_type:law_party_council
		}
		custom_tooltip = "The law will get an amendment dependent of your instituional IG"
		random_law = {
			limit = {
				type = law_type:law_counciliary
			}

		}

	}

	ai_will_do = {
		exists = ruler
		ruler = {
			has_socialist_ideology = yes
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0
	}
}

REPLACE_OR_CREATE:law_parliamentary = {
	group = lawgroup_legislative_principle

	icon = "gfx/interface/icons/law_icons/parliamentary_republic.dds"

	progressiveness = 100

	unlocking_technologies = {
		democracy
	}

	modifier = {
	}

	on_activate = {
		if = {
			limit = {
				game_date > 1836.1.1
				country_has_voting_franchise = yes
			}
			initialize_electoral_confidence = yes
		}
		scripted_effect_parties_emergence = yes
		inherit_free_elections_effect = yes
		if = {
			limit = {
				has_law_or_variant = law_type:law_aristocratic
			}
			activate_law = law_type:law_noble_assembly
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_institutional
			}
			activate_law = law_type:law_managed_democracy
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_plutocratic
			}
			activate_law = law_type:law_wealth_voting
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_meritocratic
			}
			activate_law = law_type:law_literacy_voting
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_egalitarian
			}
			activate_law = law_type:law_universal_voting
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_partisan
			}
			activate_law = law_type:law_party_assembly
		}
	}

	ai_will_do = {
		exists = ruler
		OR = {
			ruler = {
				OR = {
					has_ideology = ideology:ideology_republican_leader
					has_ideology = ideology:ideology_radical
					has_ideology = ideology:ideology_positivist
					has_socialist_ideology = yes
				}
			}
			any_interest_group = {
				is_in_government = yes
				is_powerful = yes
				leader = {
					OR = {
						has_ideology = ideology:ideology_republican_leader
						has_ideology = ideology:ideology_radical
						has_ideology = ideology:ideology_positivist
						has_socialist_ideology = yes
					}
				}
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_parliamentary_republic
				has_strategy = ai_strategy_progressive_agenda
				power_bloc ?= { has_identity = identity:identity_ideological_union }
				scope:target_country = { # Is their law different enough from what we want to impose?
					NOT = { has_law_or_variant = law_type:law_presidential_republic }
				}
			}

			add = base_impose_law_weight
		}
	}
}
