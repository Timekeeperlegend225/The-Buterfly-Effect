REPLACE_OR_CREATE:law_institutional_landowners = {
	group = lawgroup_institutional_group

	icon = "gfx/interface/icons/ig_icons/landowners.dds"

	progressiveness = -50

	modifier = {
		interest_group_ig_landowners_pol_str_mult = 0.05	
	}

	on_activate = {
		if = {
			limit = {
				has_law_or_variant = law_type:law_institutional_rule
			}
			active_law:lawgroup_governance_principles ?= { 
				every_scope_amendment = {
					limit = {
						OR = {
							type = amendment_type:TBE_amendement_landowners_rule
							type = amendment_type:TBE_amendement_devout_rule
							type = amendment_type:TBE_amendement_armed_forces_rule
							type = amendment_type:TBE_amendement_petty_bourgeoisie_rule
							type = amendment_type:TBE_amendement_intelligentsia_rule
							type = amendment_type:TBE_amendement_industrialists_rule
							type = amendment_type:TBE_amendement_trade_unions_rule
							type = amendment_type:TBE_amendement_rural_folk_rule
						}
					}
					remove_amendment = yes
				}
			}
			active_law:lawgroup_governance_principles ?= { 
				add_amendment = {
					type = TBE_amendement_landowners_rule
					sponsor = ROOT.ig:ig_landowners
					cooldown = 0
				}
			}
		}
	}

	ai_will_do = {
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
	}

	ai_impose_chance = {
		value = 0
	}
}
