REPLACE:movement_labor = { # Explicitly non-socialist labour movement
	category = movement_category_ideological

	ideology = ideology_proletarian_movement
	character_ideologies = {
		ideology_reformer
		ideology_radical
		ideology_social_democrat
		ideology_utopian
	}

	creation_trigger = {
		has_technology_researched = labor_movement
		# Intentionally not modernisation-gated, if someone manages to proletarianise without modernising this should have consequences
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_labor
							}
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_reformer
			has_ideology = ideology:ideology_radical
			has_ideology = ideology:ideology_social_democrat
			has_ideology = ideology:ideology_utopian
			is_interest_group_type = ig_trade_unions
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_social_democrat
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_reformer
					has_ideology = ideology:ideology_radical
				}
			}
			multiply = {
				value = 2
			}
		}
		if = { # Labour movement is the natural habitat of non-socialist trade unions characters
			limit = {
				is_interest_group_type = ig_trade_unions
				has_socialist_ideology = no
			}
			multiply = {
				value = 2
			}
		}
		if = { # Communists will support the labour movement if they really, really have to
			limit = {
				owner ?= {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_anarchist
							is_political_movement_type = movement_socialist
							is_political_movement_type = movement_communist
						}
					}
				}
				OR = {
					has_ideology = ideology:ideology_anarchist
					has_ideology = ideology:ideology_vanguardist
					has_ideology = ideology:ideology_communist
				}
			}
			multiply = {
				value = 0.01
			}
		}
	}

	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_engineers
		movement_support_academics
		movement_support_laborers
		movement_support_machinists
		movement_support_clerks
	}
	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
					}
				}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = upper
			}
			multiply = {
				value = 0.10
				desc = "UPPER_NO_ICON"
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 0.5
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = engineers
				}
				add = {
					value = 3
					desc = "POP_ENGINEERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 6
					desc = "POP_ACADEMICS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 25
					desc = "POP_LABORERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 30
					desc = "POP_MACHINISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 10
					desc = "POP_CLERKS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "LOWER_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				owner = {
					has_technology_researched = socialism
				}
			}
			if = {
				limit = {
					owner = {
						NOT = { has_law_or_variant = law_type:law_public_schools }
					}
				}
				multiply = 1.25
			}
			if = {
				limit = {
					owner = {
						NOT = { has_law_or_variant = law_type:law_public_health_insurance }
					}
				}
				multiply = 1.25
			}
			if = {
				limit = {
					owner = {
						has_law_or_variant = law_type:law_no_workers_rights
					}
				}
				multiply = 1.25
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol # It is intentional that a labourer being well-off doesn't decrease support for this.
			}
			add = { # Increases likelihood of support if impoverished.
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		if = { # If a more radical movement exists, it'll poach radicals from this one.
			limit = {
				owner ?= {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_socialist
							is_political_movement_type = movement_communist
							is_political_movement_type = movement_anarchist
						}
					}
				}
			}
			multiply = {
				desc = "POP_RADICAL_MOVEMENT_LOSSES"
				value = 1
				subtract = pop_radical_fraction
			}
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = state_urbanization_rate
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate
		}
	}

	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_social_security }

				add = {
					value = 3
					subtract = "institution_investment_level(institution_social_security)"
					multiply = 0.025
					desc = "INSTITUTION_FUNDING_LEVEL_social_security"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_social_security"
				}
			}

			if = {
				limit = { has_institution = institution_workplace_safety }

				add = {
					value = 3
					subtract = "institution_investment_level(institution_workplace_safety)"
					multiply = 0.025
					desc = "INSTITUTION_FUNDING_LEVEL_workplace_safety"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_workplace_safety"
				}
			}
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								is_political_movement_type = movement_labor
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_REVOLUTION"
				value = 0.15
			}
		}
	}
}

REPLACE:movement_socialist = {
	category = movement_category_ideological

	ideology = ideology_socialist_movement
	character_ideologies = {
		ideology_communist
		ideology_anarchist
		ideology_social_democrat
		ideology_vanguardist
	}

	creation_trigger = {
		has_technology_researched = socialism
		country_should_exclusively_focus_on_modernization = no
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_socialist
							}
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	disband_trigger = {
		any_interest_group = {
			OR = {
				has_variable = chose_anarchism
				has_variable = chose_communism
				has_variable = chose_vanguardism
			}
		}
	}
	on_disbanded = {}

	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_communist
			has_ideology = ideology:ideology_anarchist
			has_ideology = ideology:ideology_social_democrat
			has_ideology = ideology:ideology_vanguardist
			has_ideology = ideology:ideology_utopian
			AND = {
				is_interest_group_type = ig_trade_unions
				has_ideology = ideology:ideology_humanitarian
			}
		}
	}
	character_support_weight = {
		value = 200
		if = { # Very vulnerable to fragmentation once the more focused movements emerge
			limit = {
				OR = {
					has_ideology = ideology:ideology_vanguardist
					has_ideology = ideology:ideology_communist
				}
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_communist
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_anarchist
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_anarchist
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_social_democrat
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_labor
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
	}

	pop_support_trigger = {
		NOR = {
			is_pop_type = capitalists
			is_pop_type = aristocrats
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_radicalization
		movement_support_academics
		movement_support_laborers
		movement_support_machinists
		movement_support_soldiers_conscripts
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
					}
				}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 0.10
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 3
					desc = "POP_ACADEMICS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 18
					desc = "POP_LABORERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 24
					desc = "POP_MACHINISTS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "LOWER_NO_ICON"
				}
			}

			if = {
				limit = {
					AND = {
						is_pop_type = soldiers
						pop_employment_building = building_conscription_center
					}
				}
				multiply = {
					value = 1.5
					desc = "CONSCRIPTS"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = lower
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}

			if = {
				limit = {
					owner = {
						NOR = {
							has_law_or_variant = law_type:law_council_republic
							has_law_or_variant = law_type:law_universal_suffrage
		}
					}
				}
				multiply = 2.0
			}
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = lower
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = { # Multiplied by 1.5x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			divide = 2
			add = 1
		}

		if = {
			limit = {
				owner = {
					any_political_movement = {
						count >= 2

						OR = {
							is_political_movement_type = movement_labor
							is_political_movement_type = movement_communist
							is_political_movement_type = movement_anarchist
	}
						political_movement_support >= 0.10
					}
				}
			}
			multiply = 0.05
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = state_urbanization_rate

			if = {
				limit = { is_capital = yes }
				if = {
					limit = { owner ?= { has_law_or_variant = law_type:law_council_republic } }
					multiply = 10.0
				}
				else = {
					multiply = 0.1
				}
			}
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate

			if = {
				limit = { is_capital = yes }
				if = {
					limit = { owner ?= { has_law_or_variant = law_type:law_council_republic } }
					multiply = 10.0
				}
				else = {
					multiply = 0.1
				}
			}
		}
	}

	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_social_security }

				add = {
					value = 3
					subtract = "institution_investment_level(institution_social_security)"
					multiply = 0.025
					desc = "INSTITUTION_FUNDING_LEVEL_social_security"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_social_security"
				}
			}

			if = {
				limit = { has_institution = institution_workplace_safety }

				add = {
					value = 3
					subtract = "institution_investment_level(institution_workplace_safety)"
					multiply = 0.025
					desc = "INSTITUTION_FUNDING_LEVEL_workplace_safety"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_workplace_safety"
				}
			}

			add = {
				desc = "IS_BELOW_15_SOL"
				value = 15
				subtract = average_sol
				multiply = 0.03
				min = 0
			}

			if = {
				limit = {
					has_law_or_variant = law_type:law_factory_councils
				}

				if = {
					limit = {
						country_is_abolishing_factory_councils = yes
					}

					add = {
						value = 0.25
						desc = "FACTORY_COUNCILS_BEING_ABOLISHED"
					}
				}
			}
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								is_political_movement_type = movement_socialist
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_REVOLUTION"
				value = 0.15
			}
		}
	}
}

							is_building_type = building_naval_base
							state = { is_slave_state = yes }
						}
						add = this.level
					}
				}

				divide = {
					value = owner.navy_size
					min = 1
				}

				subtract = 0.5
				multiply = -0.5
				min = 0
				max = 0.25
				desc = "NAVY_FRACTION_IN_SLAVE_STATES"
			}
		}

		if = {
			limit = {
				owner ?= { has_journal_entry = je_acw_countdown }
			}

			add = {
				value = 0.2
				desc = "je_acw_countdown"
			}
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								is_political_movement_type = movement_pro_slavery
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_REVOLUTION"
				value = 0.15
			}
		}
	}
